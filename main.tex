\section{Introducción}

\begin{frame}{Introducción a la plantilla}
   
   Esta es una plantilla de ejemplo para crear presentaciones en LaTeX con el estilo visual de la \textbf{Universidad de La Laguna (ULL)}.
   
   \vspace{1em}
   \begin{itemize}

       \item Los colores principales están adaptados a la guía de estilo.

   \end{itemize}
\end{frame}

\section{Algoritmos y Pseudocódigo}

\begin{frame}[s]
   \frametitle{Ejemplo de Algoritmo (Dijkstra)}
   \begin{adjustbox}{max totalsize={!}{0.8\textheight},center}
      \begin{algorithm}[H]
         \caption{Algoritmo de Dijkstra}
         \KwIn{Grafo \var{G}, nodo origen \var{u}}
         \KwOut{Array de distancias mínimas \var{dist}$[\cdot]$}
         \BlankLine
         \tcc{Inicialización}
         \var{visitados} $\leftarrow$ \{\var{u}\}\;
         \var{dist}[\var{u}] $\leftarrow$ 0\;
         \For{cada nodo \var{v} en \var{G} (excepto \var{u})}{
            \If{\var{v} es vecino de \var{u}}{
               \var{dist}[\var{v}] $\leftarrow$ coste(\var{u},\var{v})\;
            }
            \Else{
               \var{dist}[\var{v}] $\leftarrow$ $\infty$\;
            }
         }
         \BlankLine
         \tcc{Bucle principal}
         \While{queden nodos sin visitar}{
            \var{w} $\leftarrow$ nodo no visitado con menor \var{dist}[\var{w}]\;
            Añadir \var{w} a \var{visitados}\;
            \For{cada vecino \var{v} de \var{w} no visitado}{
               \var{dist}[\var{v}] $\leftarrow$ $\min$(\var{dist}[\var{v}], \var{dist}[\var{w}] + coste(\var{w},\var{v}))\;
            }
         }
      \end{algorithm}
   \end{adjustbox}
\end{frame}
